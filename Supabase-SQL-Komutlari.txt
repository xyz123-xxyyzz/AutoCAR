-- 3. Eklentinin bu ayarları çekebilmesi için gerekli olan RPC fonksiyonunu oluşturuyoruz:
CREATE OR REPLACE FUNCTION public.get_system_config()
RETURNS json AS $$
DECLARE
  result json;
BEGIN
  SELECT row_to_json(t) INTO result FROM public.system_config t WHERE id = 1;
  RETURN result;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 4. Fonksiyona dışarıdan (anonim olarak) erişim yetkisi veriyoruz (zaten eklenti Supabase Anon Key kullanıyor):
GRANT EXECUTE ON FUNCTION public.get_system_config() TO anon;
GRANT EXECUTE ON FUNCTION public.get_system_config() TO authenticated;
